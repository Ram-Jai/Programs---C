// average using arrays
#include<stdio.h>
int main() {
    int a[10],n,i;
    float m=0;
    printf("Enter the size of array\n");
    scanf("%d",&n);
    printf("Enter the elements into the array\n");
    for (i=0;i<n;i++)
        scanf("%d",&a[i]);
    for (i=0;i<n;i++)
        m+=a[i];
    printf("The average of all the numbers in the array is %f",m/n);
}
