//push all zeros in an array to the end

#include<stdio.h>

void main() 
{
	int n,i,j,count;			//n-size of array
	scanf("%d",&n);
	
	int a[n];				//a[n]-array
	for (i=0;i<n;i++)
		scanf("%d",&a[i]);

	for (i=0,j=n-1;i<n,j>0;i++){
		if (a[i]==0){
			a[j]=a[i]+a[j]-(a[i]=a[j]);
			j++;
		}
	}
	
	for (i=0;i<n;i++)
		printf("%d",a[i]);
		
}
