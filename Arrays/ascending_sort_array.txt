// find 2 maximum no.s and their product
#include<stdio.h>
int main() {
    int a[20],i,j,n;
    printf("Enter the number of elements in the array\n");
    scanf("%d",&n);
    for (i=0;i<n;i++)
        scanf("%d",&a[i]);
    for (i=0;i<n-1;i++)
        for (j=i+1;j<n-1;j++){
            if (a[i]>a[j])
                a[i]=a[i]+a[j]-(a[j]=a[i]);
            else if (a[i]==a[j])
                a[i+1]=a[i+1]+a[j]-(a[j]=a[i+1]);
            else
                continue;
        }    
    printf("The ascending order of the array of numbers is:\n");
    for (i=0;i<n;i++)
        printf("%d\n",a[i]);
}
